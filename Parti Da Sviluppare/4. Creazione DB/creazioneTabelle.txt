-- Creazione della tabella PAZIENTE
CREATE TABLE `ClinicaMedicaDB`.`PAZIENTE` (
    CF CHAR(16) PRIMARY KEY,
    NOME VARCHAR(16),
    COGNOME VARCHAR(16),
    EMAIL VARCHAR(64),
    TELEFONO CHAR(13),
    SESSO CHAR(1),
    INFORMAZIONI TEXT
);

-- Creazione della tabella MEDICO
CREATE TABLE `ClinicaMedicaDB`.`MEDICO` (
    REG_MEDICA CHAR(16) PRIMARY KEY,
    NOME VARCHAR(16),
    COGNOME VARCHAR(16),
    SPECIALIZZAZIONE VARCHAR(128),
    TELEFONO CHAR(13),
    SESSO CHAR(1)
);

-- Creazione della tabella SERVIZIO
CREATE TABLE `ClinicaMedicaDB`.`SERVIZIO` (
    CODICE INT PRIMARY KEY AUTO_INCREMENT,
    NOME_SERVIZIO VARCHAR(128),
    DURATA INT,
    COSTO INT
);

-- Creazione della tabella TRATTAMENTO
CREATE TABLE `ClinicaMedicaDB`.`TRATTAMENTO` (
    CODICE INT PRIMARY KEY AUTO_INCREMENT,
    NOME_TRATTAMENTO VARCHAR(128),
    DURATA INT,
    AVVERTENZE TEXT,
    OBBIETTIVI TEXT
);

-- Creazione della tabella APPUNTAMENTO
CREATE TABLE `ClinicaMedicaDB`.`APPUNTAMENTO` (
    CODICE INT PRIMARY KEY AUTO_INCREMENT,
    MOTIVO VARCHAR(128),
    ORA CHAR(5), -- XX:XX (5 caratteri)
    DATA CHAR(10), -- GG/MM/AAAA (10 caratteri)
    CF_PAZIENTE CHAR(16),
    TRATT_PROPOSTO INT,
    COD_SERVIZIO INT,
    FOREIGN KEY (CF_PAZIENTE) REFERENCES PAZIENTE(CF),
    FOREIGN KEY (COD_SERVIZIO) REFERENCES SERVIZIO(CODICE)
);

-- Creazione della tabella FATTURA
CREATE TABLE `ClinicaMedicaDB`.`FATTURA` (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    MODALITA VARCHAR(64),
    STATO VARCHAR(128),
    DATA_PAGAMENTO CHAR(10), -- GG/MM/AAAA (10 caratteri)
    COD_SERVIZIO INT,
    CF_CLIENTE CHAR(16),
    REG_MEDICO CHAR(16),
    FOREIGN KEY (COD_SERVIZIO) REFERENCES SERVIZIO(CODICE),
    FOREIGN KEY (CF_CLIENTE) REFERENCES PAZIENTE(CF),
    FOREIGN KEY (REG_MEDICO) REFERENCES MEDICO(REG_MEDICA)
);
